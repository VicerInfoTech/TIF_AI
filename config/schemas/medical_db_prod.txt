-- Example DDL for medical_db_prod
CREATE DATABASE "Attendance_sys" WITH TEMPLATE = template0 ENCODING = 'UTF8' LOCALE_PROVIDER = libc LOCALE = 'English_United States.1252';

CREATE SCHEMA public;

CREATE TABLE public.attendance_logs (
    id integer NOT NULL,
    employee_id integer,
    logged_at timestamp without time zone DEFAULT now(),
    camera_id text,
    location text,
    method text
);


CREATE SEQUENCE public.attendance_logs_id_seq
    AS integer
    START WITH 1
    INCREMENT BY 1
    NO MINVALUE
    NO MAXVALUE
    CACHE 1;

CREATE TABLE public.employees (
    id integer NOT NULL,
    name text NOT NULL,
    embedding public.vector(1000)
);

CREATE SEQUENCE public.employees_id_seq
    AS integer
    START WITH 1
    INCREMENT BY 1
    NO MINVALUE
    NO MAXVALUE
    CACHE 1;


CREATE INDEX employees_embedding_idx ON public.employees USING hnsw (embedding public.vector_l2_ops);
